---
title: size
desc: setting different size of loading( small, middle, large ). Default is "middle".
---

```vue
<template>
  <vue-fun-loading size="small" />
  <vue-fun-loading />
  <vue-fun-loading size="large" />
</template>

<script>
import VueFunLoading from 'vue-fun-loading'

export default {
  components: {
    VueFunLoading
  }
}
</script>
```
